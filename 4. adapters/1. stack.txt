#include <stack>
#include <iostream>
using namespace std;

int main()
{
	stack<int> s1, s2;

	s1.push(10); // добавляет элемент в вершину стека
	s1.push(20);
	s1.push(30);

	// возвращает количество элементов в стеке
	cout << "The stack length is " << s1.size() << "." << endl;

	// возвращает ссылку на элемент в вершине стека
	cout << "The element at the top of the stack is " << s1.top() << "." << endl;

	s1.pop(); // удаляет элемент из вершины стека

	cout << "After a pop, the stack length is " << s1.size() << "." << endl;

	cout << "After a pop, the element at the top of the stack is " << s1.top() << "." << endl;

	if (s1.empty()) // проверяет, пустой ли стек
		cout << "The stack s1 is empty." << endl;
	else
		cout << "The stack s1 is not empty." << endl;

	if (s2.empty())
		cout << "The stack s2 is empty." << endl;
	else
		cout << "The stack s2 is not empty." << endl;

}
